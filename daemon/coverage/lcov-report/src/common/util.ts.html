export const randomString = (num_chars: number) => {

const sortKeysInObject = (x: any): any => {
    if (x instanceof Buffer) {
        return x;
    }
    else if (x instanceof Object) {

            return x.map(a => (sortKeysInObject(a)));

            const keys = Object.keys(x).sort();
            let ret: any = {};

}
